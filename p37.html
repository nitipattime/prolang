ความเชื่อมโยงของตัวดำเนินการ

เมื่อประโยคคำสั่งมี ลำดับการประเมินผลนิพจน์จะต้องทำตามความเชื่อมโยงที่ภาษาการโปรแกรมกำหนด ซึ่งจะมีการกำหนดลำดับอยู่สองแบบคือ หรือ

ตัวอย่างภาษา C++ กำหนดให้ Incrementer มีความเชื่อมโยงจากขวาไปซ้าย 

#include 
        using namespace std;
        int main()
        {
            int a = 10, b = 20 ;
            a = ++a + ++b;      //ต้องทำการเพิ่มค่า(บวกเพิ่ม 1)ให้กับ b ก่อน แล้วเพิ่มค่าให้กับ a (บวกเพิ่ม 1)
            cout << a ;
            cout << "\n";
            cout << b ;
            return 0;
        }

https://docs.google.com/spreadsheets/d/1EvDkSn4p7BerTPtmWm-X2uBWAklbQIM5qtb2hgVI3Sg/edit?usp=sharing โดยสรุป ลำดับการประเมินผลนิพจน์ จะพิจารณา Precedence Rule ก่อนพิจารณา Associativity ตัวอย่างโค้ด C++ ที่การเขียนคำสั่งมีความซับซ้อนในการกำหนดโอเปอเรเตอร์ของนิพจน์ 

#include 
        using namespace std;
        int main()
        {
            int a = 10, b = 20 ;
            a = ++a + -++b ;         //- มี Precedence rule เท่ากับ ++ และทั้งสอง มี Associativity right-to-left
            if (a < b || b < a){     //< มี Precedence rule สูงกว่า || และ <  มี Associativity right-to-left
              cout<< "a is ";
              cout << a ;
            }
            else{
              cout<< "b is ";
              cout << b ;
            }
            return 0;
        }
