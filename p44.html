ชนิดข้อมูลที่ผู้ใช้กำหนด

ในภาษา C เราสามารถกำหนดชนิดข้อมูลเชิงโครงสร้าง ซึ่งมีคุณลักษณะคล้ายการกำหนดชนิดข้อมูลแบบเรคคอร์ด 

#define  STRSIZ 10
        typedef struct {
              char   name[STRSIZ];
             double diameter;
             int  moons;
            double orbit_time, rotation_time;
        }  planet_t;

การประกาศตัวแปรซึ่งมีชนิดข้อมูลเชิงโครงสร้าง planet_t สามารถกำหนดได้ดังนี้ planet_t current_planet, previous_planet, blank_planet = {"", 0,0,0,0}; ตัวแปร blank_planet จะมีโครงสร้างข้อมูลดังภาพ

การอ้างอิงถึงสมาชิกในข้อมูลเชิงโครงสร้าง สามารถกำหนดได้ดังนี้ ชื่อตัวแปร.ฟิลด์ข้อมูล เช่น

current_planet.name, current_planet.diameter

เป็น User-Defined data type หรือชนิดข้อมูลที่ผู้ใช้กำหนด เมื่อต้องการกำหนดพื้นที่ในเมมโมรีที่ต้องถูกจัดสรรเพื่อใช้งานร่วมกันระหว่างหลายตัวแปร โดยในการใช้งานจะมีเพียงหนึ่งตัวแปร (ใน union) เท่านั้นที่สามารถใช้พื้นที่ในเมมโมรีได้ พื้นที่ของตัวแปรที่มีชนิดข้อมูลแบบ union จะมีขนาดสูงสุดซึ่งเป็นไปตามชนิดข้อมูลที่มีขนาดใหญ่สุดที่กำหนดในชนิดข้อมูล union 

#include 
        union Job
        {
           float salary;     //sizeof float = 4 bytes
           int workerNo;  //sizeof int = 4 bytes
        } j;     //ตัวแปร j มีชนิดข้อมูลเป็น union
        
        int main()
        {
            j.salary = 12.3;   //ใช้หน่วยความจำ
           j.workerNo = 100;   //พื้นที่ในหน่วยความจำถูกใช้แทนสำหรับเก็บค่า workerNo
        
           printf("Salary = %.1f  size = %d\n", j.salary, sizeof(j.salary));
           printf("Number of workers = %d size = %d\n ", j.workerNo,sizeof(j.workerNo));
           printf("Union j size = %d\n", sizeof(j));
           return 0;
        }

ในโค้ดนั้นคำสั่งการใช้พื้นที่เพื่อกำหนดให้ workerNo เป็นคำสั่งหลังสุด การกำหนดค่า salary ในหน่วยความจำถูกแทนที่ด้วยค่า workerNo เพราะเงื่อนไขของการใช้พื้นที่ในหน่วยความจำนั้นกำหนดว่าจะมีเพียงหนึ่งตัวแปรเท่านั้นที่ใช้พื้นที่ได้
