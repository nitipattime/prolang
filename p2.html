เรียนหลักการภาษาโปรแกรมจะได้อะไร ?

    สิ่งที่จะได้จากการเรียนหลักการของภาษาโปรแกรม อาทิ
  1. ทำให้ผู้เรียนสามารถแสดงแนวคิดในการแก้ปัญหาได้อย่างหลากหลายวิธี เมื่อมีความรู้ในหลักการต่างๆ ก็จะสามารถนำความรู้นั้นมาประยุกต์ในการแก้ไขปัญหาได้
  2. ทำให้ผู้เรียนสามารถเลือกใช้ภาษาสำหรับการเขียนโปรแกรมที่เหมาะสมในการแก้ไขปัญหาหรือพัฒนาระบบงานได้
  3. ทำให้ผู้เรียนเรียนรู้ภาษาสำหรับการเขียนโปรแกรมใหม่ ๆ ได้ง่ายขึ้น สามารถเรียนรู้ด้วยความเข้าใจโดยใช้เวลาที่สั้นลง
  4. ทำให้ผู้เรียนมีความเข้าใจที่ดีขึ้นในการเขียนโปรแกรม เช่น ทราบลำดับและวิธีในการประมวลผลคำสั่งในโปรแกรม
  5. ทำให้ผู้เรียนสามารถหาข้อบกพร่องของโปรแกรม (bug) ได้เร็วขึ้น และสามารถแก้ไขคำสั่งให้ถูกต้องได้
  6. ทำให้ผู้เรียนมีความรู้และความเข้าใจมากขึ้นในภาษาที่ที่เคยเรียนไปแล้ว
  7. ทำให้ผู้เรียนสามารถเขียนโปรแกรมที่มีการประมวลผลที่มีประสิทธิภาพดีขึ้น โดยสามารถเลือกใช้คำสั่งที่เหมาะสมในการประมวลผล
หลังจากที่ได้เลือกภาษาสำหรับการเขียนโปรแกรมเพื่อพัฒนาระบบงานหรือโปรแกรมแล้ว ผู้เรียนจะสามารถเข้าใจ อาจพิจารณาจากข้อต่อไปนี้
  1. โปรแกรม
  4. โปรแกรม เช่น การคำนวณต่าง ๆ มีความถูกต้อง
  5. โปรแกรม เช่น ไม่ใช้หน่วยความจำสิ้นเปลืองจนกระทั่งประมวลผลต่อไม่ได้
  6. โค้ดในโปรแกรมซึ่งจะเป็นประโยชน์ในการปรับปรุงในระหว่างการดูแลรักษาโปรแกรมในระยะยาว
ให้ท่านดูโค้ดต่อไปนี้ ซึ่งเขียนในภาษา C แล้วลองวิจารณ์โปรแกรมเทียบกับคุณลักษณะของโปรแกรมที่ดี 
        #include 
         <stdio.h>
       int main(void) {
       int seg[10] = {6,2,5,5,4,5,6,3,7,6};
       int d1, d2, d3, d4, m=0, td, ts;
       for (d1=0; d1<2; d1++)
       for (d2=0; d2<10; d2++)
       for (d3=0; d3<6; d3++) for (d4=0; d4<10; d4++)
       if ((!((d1==0)&&(d2==0))) && (!((d1==1)&&(d2>2)))) {
       if (d1==0) { ts = seg[d2] + seg[d3] + seg[d4]; td = d2 + d3 + d4;
       if (ts == td) { m++;  printf(" %1d:%1d%1d\n",d2,d3,d4); }
       } else {
       ts = seg[d1] + seg[d2] + seg[d3] + seg[d4];
       td = d1 + d2 + d3 + d4;
       if (ts == td) { m++;
       printf("%1d%1d:%1d%1d\n",d1,d2,d3,d4); }
       } }
       return 0; }
