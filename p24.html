ตัวแปรโกลบอล

มีนิยามแตกต่างกันไปในภาษาการโปรแกรมต่างๆ สิ่งที่ต้องพิจารณาคือการมองเห็น และการแก้ไขค่าตัวแปรโกลบอล ภาษาส่วนใหญ่กำหนดว่า Global variable คือ ตัวแปรที่ประกาศใน หรือมีหน่วยเป็นโปรแกรม
คือ หน่วยโปรแกรมซึ่งครอบคลุมหน่วยอื่นๆ

globvar = 0                   //Global scope  ตัวแปรที่ประกาศใน Global scope คือ Global variable
        
        def set_globvar_to_one():     //Local scope
            global globvar            //ตัวแปร Global variable ใน Python จะ่สามารถมองเห็นในฟังก์ชันในกรณีที่เข้าถึงเพื่ออ่าน
            globvar = 1               //แต่เมื่อกำหนด global จึงจะสามารถกำหนดค่าได้
        
        def print_globvar():          //Local scope
            print(globvar)            //Local scope
        
        
        set_globvar_to_one()        //Global scope
        print_globvar()             //Global scope

ตัวอย่างการนิยามตัวแปรโกลบอลใน
หมายถึง ตัวแปรที่สามารถมองเห็นได้จากทุก ๆ ส่วนในโปรแกรม เช่นในภาษา C เมื่อทำการประกาศตัวแปรไว้เหนือฟังก์ชันหลัก (main) ตัวแปรเหล่านั้นจะเป็นตัวแปรโกลบอลที่สามารถมองเห็นได้จากทุก ๆ ส่วนในโปรแกรม 

#include 
        int d =  50;        /* Global variable */
        int main () {
        
          /* local variable declaration */
          int a, b;
          int c;
        
          /* actual initialization */
          a = 10;
          b = 20;
          c = a + b;
        
          printf ("value of a = %d, b = %d and c = %d\n", a, b, c);
        
          return 0;
        }

ตัวอย่างการนิยามตัวแปรโกลบอลใน
หมายถึง ตัวแปรภายนอกฟังก์ชันและถูกกำหนดการประกาศในบริเวณของเมนโปรแกรม ซึ่งการมองเห็นจากฟังก์ชันจะไม่สามารถทำได้ จะต้องใช้คีย์เวิร์ด global หรือ GLOBAL จึงจะสามารถมองเห็นได้และแก้ไขค่าข้อมูลได้ต่อไป

ตัวอย่างการนิยามตัวแปรโกลบอลใน
หมายถึง ตัวแปรภายนอกฟังก์ชันและถูกกำหนดการประกาศในบริเวณของเมนโปรแกรม ซึ่งสามารถมองเห็นได้จากฟังก์ชัน โดยสามารถอ่านค่าได้ (read) แต่ไม่สามารถแก้ไข/เขียนค่าได้ จะต้องกำหนดคีย์เวิร์ด global เพื่อการเขียนเท่านั้น ตัวแปร Global ไม่ใช่ตัวแปรที่สามารถเห็นได้จากทุกๆส่วนของโปรแกรม ขึ้นกับข้อกำหนดของภาษา ตัวอย่างภาษา Python ตัวแปร Global variable สามารถอ่านได้จากภายในฟังก์ชัน แต่ไม่สามารถแก้ไขได้ หากต้องการแก้ไขต้องกำหนดด้วยคีย์เวิร์ด global ตัวอย่างการอ่านค่า Global variable อย่างเดียว

c = 1 # global variable
        def add():
            print(c)  #read global variable
        add()
        print(c)   #global variable

ตัวอย่างเมื่อต้องการเขียนค่า Global variable

c = 1 # global variable
        def add():
            global c    #global  variable
            c = c + 2   #edit global variable
            print(c)
        add()
        print(c)   #global variable is edited

หากไม่มีการกำหนด global ดังนั้น 

c = 1 # global variable
        def add():
            c = 0     #local variable not global variable
            c = c + 2
            print(c)
        add()
        print(c)   #global variable
